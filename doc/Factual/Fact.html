def input(value, opts={})
  return false if value.nil?

  hash = opts.merge({
    :subjectKey => @subject_key.first,
    :values     => [{
      :fieldId    => @field['id'],
      :value      => value }]
  })

  @adapter.input(@table_key, hash)
  return true
end
